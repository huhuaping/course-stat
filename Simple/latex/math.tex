\HeaderA{math}{Standardized math scores}{math}
\keyword{datasets}{math}
\begin{Description}\relax
Standardized math scores
\end{Description}
\begin{Usage}
\begin{verbatim}data(math)\end{verbatim}
\end{Usage}
\begin{Format}\relax
The format is:
num [1:30] 44 49 62 45 51 59 57 55 70 64 ...
\end{Format}
\begin{Source}\relax
From Kitchens Exploring Statistics
\end{Source}
\begin{Examples}
\begin{ExampleCode}
data(math)
hist(math)
\end{ExampleCode}
\end{Examples}
